local function buttonHover(button, originalTransparency)
    button.MouseEnter:Connect(function()
        TweenService:Create(button, TweenInfo.new(0.2), {
            BackgroundTransparency = originalTransparency - 0.1,
            Size = UDim2.new(button.Size.X.Scale + 0.02, 0, 1.1, 0)
        }):Play()
    end)
    
    button.MouseLeave:Connect(function()
        TweenService:Create(button, TweenInfo.new(0.2), {
            BackgroundTransparency = originalTransparency,
            Size = UDim2.new(0.45, 0, 1, 0)
        }):Play()
    end)
end

buttonHover(YesButton, 0.3)

local countdownTimer = nil
local countdownTime = 5
local isNotificationActive = false

local function showNotification()
    if isNotificationActive then return end
    isNotificationActive = true
    
    NotificationFrame.Position = UDim2.new(0.5, -200, 0.2, -90)
    NotificationFrame.BackgroundTransparency = 1
    NotificationFrame.Visible = true
    
    TweenService:Create(NotificationFrame, TweenInfo.new(0.5, Enum.EasingStyle.Back, Enum.EasingDirection.Out), {
        Position = UDim2.new(0.5, -200, 0.3, -90),
        BackgroundTransparency = 0.3
    }):Play()
    
    countdownTime = 5
    TimerLabel.Text = "Auto-disables in " .. countdownTime .. " seconds"
    
    if countdownTimer then
        countdownTimer:Disconnect()
    end
    
    countdownTimer = game:GetService("RunService").Heartbeat:Connect(function(delta)
        countdownTime = countdownTime - delta
        if countdownTime <= 0 then
            countdownTimer:Disconnect()
            TimerLabel.Text = "Auto-disabled"
            
            task.wait(0.5)
            NoButton.MouseButton1Click:Fire()
        else
            TimerLabel.Text = "Auto-disables in " .. math.floor(countdownTime) .. " seconds"
        end
    end)
end

local function hideNotification()
    if not isNotificationActive then return end
    
    if countdownTimer then
        countdownTimer:Disconnect()
        countdownTimer = nil
    end
    
    TweenService:Create(NotificationFrame, TweenInfo.new(0.5, Enum.EasingStyle.Back, Enum.EasingDirection.In), {
        Position = UDim2.new(0.5, -200, 0.2, -90),
        BackgroundTransparency = 1
    }):Play()
    
    task.wait(0.5)
    NotificationFrame.Visible = false
    isNotificationActive = false
end

YesButton.MouseButton1Click:Connect(function()
    print("[Rank Tags] User chose YES")
    hideNotification()
    
    local success, error = pcall(function()
        loadstring(game:HttpGet("https://raw.githubusercontent.com/SuperHackerYT/skibidi/refs/heads/main/TagRank.txt"))()
    end)
    
    if not success then
        warn("Failed to load Rank Tags:", error)
    end
end)

end)

NotificationFrame.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        local dragStart = input.Position
        local frameStart = NotificationFrame.Position
        
        local connection
        connection = input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then
                connection:Disconnect()
            end
        end)
        
        local dragConnection = game:GetService("UserInputService").InputChanged:Connect(function(moveInput)
            if moveInput.UserInputType == Enum.UserInputType.MouseMovement then
                local delta = moveInput.Position - dragStart
                NotificationFrame.Position = UDim2.new(
                    frameStart.X.Scale,
                    frameStart.X.Offset + delta.X,
                    frameStart.Y.Scale,
                    frameStart.Y.Offset + delta.Y
                )
            end
        end)
        
        task.wait()
        dragConnection:Disconnect()
    end
end)

local function showRankTagNotification()
    showNotification()
end
